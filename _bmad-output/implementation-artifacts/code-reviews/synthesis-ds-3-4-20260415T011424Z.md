## Changes Applied

**File**: `firmware/data-src/dashboard.js`
**Change**: `finalizeModeSwitch` — defer `modeSwitchInFlight = false` until `loadDisplayModes()` resolves in the `!skipReload` branch

**Before**:
```javascript
function finalizeModeSwitch(newActiveId, previousActiveId, skipReload) {
    modeSwitchInFlight = false;
    function doFocus() { ... }
    if (skipReload) {
      doFocus();
    } else {
      clearSwitchingStates();
      loadDisplayModes().then(doFocus);
    }
  }
```
**After**:
```javascript
function finalizeModeSwitch(newActiveId, previousActiveId, skipReload) {
    function doFocus() { ... }
    if (skipReload) {
      modeSwitchInFlight = false;   // DOM already rebuilt — safe to unlock immediately
      doFocus();
    } else {
      clearSwitchingStates();
      loadDisplayModes().then(function() {
        modeSwitchInFlight = false;
        doFocus();
      }, function() {
        modeSwitchInFlight = false;
      });
    }
  }
```
