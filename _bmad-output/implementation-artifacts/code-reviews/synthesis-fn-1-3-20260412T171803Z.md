**Change 6**: Fix clearOTAUpload to prevent aborting completed updates
**Before**:
```cpp
void clearOTAUpload(AsyncWebServerRequest* request) {
    for (auto it = g_otaUploads.begin(); it != g_otaUploads.end(); ++it) {
        if (it->request == request) {
            // CRITICAL: abort in-progress update on cleanup
            if (it->started) {
                Update.abort();
                LOG_I("OTA", "Upload aborted during cleanup");
            }
            g_otaUploads.erase(it);
            return;
        }
    }
}
```
**After**:
```cpp
void clearOTAUpload(AsyncWebServerRequest* request, bool abortIfStarted = true) {
    for (auto it = g_otaUploads.begin(); it != g_otaUploads.end(); ++it) {
        if (it->request == request) {
            // Only abort if explicitly requested (not for successful completion)
            if (it->started && abortIfStarted) {
                Update.abort();
                LOG_I("OTA", "Upload aborted during cleanup");
            }
            g_otaUploads.erase(it);
            return;
        }
    }
}
```
And update success handler:
```cpp
clearOTAUpload(request, false); // Don't abort - update completed successfully
```
